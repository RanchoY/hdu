#include<stdio.h>

//子串,长度,存next数组的
void cal_next(char *str,int * next, int len)
{
    //初始化
    int k = -1; //k表示最长前缀最后一个字母的下标,-1表示没有
    next[0] = -1;

    for(int i=1; i<len; i++)
    {
        while( k>-1 &&  str[k+1] != str[i])
        {
            k = next[k];
        }

        if(str[k+1] == str[i])
        {
            k ++; //前缀又找到了一个所以加1
        }
        next[i] = k;
    }
}

int kmp(char *str, int slen, char *ptr, int plen)
{
    int *next = new int[plen];
    cal_next( ptr, next, plen);
    int k = -1;

    for(int i=0; i<slen; i++)
    {
        while( k>-1 && ptr[k+1] != str[i])
        {
            k = next[k];
        }

        if(ptr[k+1] == str[i])
        {
            k ++;
        }

        if(k == plen -1)
        {
            return i - plen + 1; //匹配到了所有子字符串(pstr),fanhui1
        }
    }

    return -1;
}

int main()
{
    char *str = "bacbababadababacambabacaddababacasdsd";
    char *ptr = "ababaca";

    int result = kmp(str, 36, ptr, 7);
    printf("结果是%d\n", result);

}
